; Especificación léxica: define tokens y patrones regex
(define lexical
  '(
    (comment ("%" (arbno (not #\newline))) skip) ; Comentarios: % hasta newline (se ignoran)
    (whitespace (whitespace) skip)             ; Espacios en blanco (se ignoran)
    (number (digit (arbno digit)) number)      ; Números positivos: dígitos
    (number ("-" digit (arbno digit)) number)  ; Números negativos: - seguido de dígitos
    (identifier (letter (arbno (or letter digit))) symbol) ; Identificadores: letra seguida de letras/dígitos
  ))

; Gramática para el parser (producciones y constructores del AST)
(define grammar
  '(
    (program (expression) a-program)             ; Programa: una expresión
    (expression (identifier) var-exp)           ; Expresión: variable
    (expression (number) lit-exp)               ; Expresión: literal numérico
    (expression ("true") true-exp)              ; Expresión: booleano true
    (expression ("false") false-exp)            ; Expresión: booleano false
    (expression (primitive "(" (separated-list expression ",") ")") prim-exp) ; Expresión primitiva con argumentos
    (expression ("if" expression "then" expression "else" expression) if-exp)
    (expression ("let" (arbno identifier "=" expression) "in" expression) let-exp)
    (expression ("proc" "(" (separated-list type-exp identifier ",") ")" expression) proc-exp)
    (expression ("letrec" (arbno type-exp identifier "(" (separated-list type-exp identifier ",") ")" "=" expression) "in" expression) letrec-exp)
    (expression ("set" identifier "=" expression) set-exp)
    (expression ("begin" expression (arbno ";" expression) "end") begin-exp)
    (expression ("(" expression (arbno expression) ")") app-exp)
    (primitive ("+") add-prim) ; Primitiva: suma
    (primitive ("-") sub-prim) ; Primitiva: resta
    (primitive ("*") prod-prim) ; Primitiva: multiplicación
    (primitive ("/") div-prim) ; Primitiva: división
    (primitive ("and") and-prim) ; Primitiva: and lógico
    (primitive ("or") or-prim) ; Primitiva: or lógico
    (primitive ("<") less-prim) ; Primitiva: menor que
    (primitive ("<=") lesseq-prim) ; Primitiva: menor o igual
    (primitive (">") more-prim) ; Primitiva: mayor que
    (primitive (">=") moreeq-prim) ; Primitiva: mayor o igual
    (primitive ("==") eq-prim) ; Primitiva: igualdad
    (primitive ("!=") neq-prim) ; Primitiva: desigualdad
    (type-exp ("int") int-exp)
    (type-exp ("bool") bool-exp)
    (type-exp ("(" (separated-list type-exp "*") "->" type-exp ")") proc-type-exp)
  ))

###
Los datatypes para el Abstract Syntax Tree (AST) se generan automáticamente a partir de las definiciones léxicas y gramaticales. Luego, se definen el escáner y el parser. El escáner convierte una cadena de programa en una lista de tokens, y el parser convierte esos tokens en un AST estructurado.

;; Generar datatypes automáticamente a partir de lexical/grammar
(sllgen:make-define-datatypes lexical grammar)

; Scanner: convierte string en lista de tokens
; scanner -> string -> lista de tokens
(define scanner
  (lambda (program)
    ((sllgen:make-string-scanner lexical grammar) program)))

; Parser: convierte string en AST usando la gramática
; parser: string -> ast
(define parser
  (lambda (program)
    ((sllgen:make-string-parser lexical grammar) program)))

###
La estructura de datos `environment` gestiona el alcance de las variables. `empty-env` representa un entorno vacío, `extend-env` añade nuevas asociaciones de identificadores a valores, y `extend-recursively-env` se usa para definir procedimientos recursivos. Las funciones `apply-env` y `apply-env-ref` se utilizan para buscar el valor o la referencia asociada a un identificador dentro del entorno.

; Definición del entorno (environment) como estructura de datos
(define-datatype environment environment?
  (empty-env) ; Entorno vacío
  (extend-env  ; Extender entorno con nuevos bindings
    (lid (list-of symbol?)) ; Lista de identificadores
    (lval vector?)         ; Lista de valores
    (old-env environment?)) ; Entorno anterior
  (extend-recursively-env
    (procname (list-of symbol?))
    (argss (list-of (list-of symbol?)))
    (bodies (list-of expression?))
    (old-env environment?)))

; Aplicar entorno: busca un símbolo en el entorno y retorna su valor
; apply-env: environment -> valor
(define apply-env
  (lambda (env var)
    (deref (apply-env-ref env var))))

; apply-env-ref: environment -> referencia
(define apply-env-ref
  (lambda (env var)
    (cases environment env
      (empty-env ()
        (eopl:error "Variable not found" var)) ; Error si no se encuentra
      (extend-env (lid vec old-env)
        (letrec ((search-var ; Búsqueda recursiva en la lista de bindings
                    (lambda (lid lval [pos 0])
                      (cond
                        [(null? lid) (apply-env-ref old-env var)] ; Buscar en entorno anterior
                        [(equal? (car lid) var) (a-ref pos vec)]   ; Voy a generar una referencia cuando encuentro la variable
                        [else (search-var (cdr lid) vec (+ pos 1))]))) ; Continuar búsqueda
          (search-var lid vec)))
      (extend-recursively-env (procnames llargs bodies old-env)
        (letrec ((search-proc
                    (lambda (procs args bodies)
                      (cond
                        [(null? procs) (apply-env-ref old-env var)]
                        [(eqv? (car procs) var) (a-ref 0 (list->vector (list (closure (car args) (car bodies) env))))]
                        [else (search-proc (cdr procs) (cdr args) (cdr bodies))]))))
          (search-proc procnames llargs bodies)))))))

La función `value?` es un predicado simplificado que indica que cualquier valor es válido. La estructura `reference` permite manejar valores mutables a través de una posición en un vector. `deref` recupera el valor de una referencia, mientras que `setref!` actualiza ese valor. El datatype `nulltype` se utiliza para representar el valor `void`.

; Predicado value?: cualquier valor es válido (implementación simplificada)
(define value? (lambda (v) #T))

(define deref
  (lambda (ref)
    (primitive-deref ref)))

;primitive-deref: reference -> valor
(define primitive-deref

(define setref!
  (lambda (ref val)
    (primitive-setref! ref val)))

###
`eval-program` es el punto de entrada principal para la evaluación. Primero, realiza un chequeo de tipos sobre la expresión del programa utilizando el entorno de tipos inicial (`init-tenv`). Si el chequeo de tipos es exitoso, procede a evaluar la expresión del programa con el entorno de ejecución inicial (`init-env`); de lo contrario, lanza un error.

; Evaluar programa: punto de entrada, evalúa la expresión del programa
; eval-program: program -> value
(define eval-program
  (lambda (pgm)
    (cases program pgm
      (a-program (exp)
        (if (type? (type-of-expression exp init-tenv)) ; Realiza chequeo de tipos antes de la evaluación
            (eval-expression exp init-env)
            (eopl:error "The evaluation of types is incorrect!"))))))

###
La función `eval-expression` evalúa recursivamente una expresión dada un entorno. Maneja diversos tipos de expresiones:
*   Variables (`var-exp`): busca el valor en el entorno.
*   Literales (`lit-exp`, `true-exp`, `false-exp`): retorna su valor.
*   Primitivas (`prim-exp`): evalúa los argumentos y aplica la operación primitiva.
*   Condicionales (`if-exp`): evalúa la condición y ejecuta la rama correspondiente.
*   `let-exp`: extiende el entorno con nuevas asociaciones antes de evaluar el cuerpo.
*   Procedimientos (`proc-exp`): crea un closure con los argumentos, cuerpo y entorno actual.
*   Aplicación de procedimientos (`app-exp`): evalúa el operador y los operandos, luego aplica el procedimiento.
*   Procedimientos recursivos (`letrec-exp`): extiende el entorno recursivamente con las definiciones de procedimientos.
*   Bloques `begin` (`begin-exp`): evalúa expresiones secuencialmente, retornando el valor de la última.
*   Asignaciones (`set-exp`): actualiza el valor de una referencia en el entorno.

; Evaluar expresión: recursivamente evalúa según el tipo de expresión
; eval-expression: expression -> value
(define eval-expression
  (lambda (exp env)
    (cases expression exp
      (var-exp (id) (apply-env env id)) ; Variable: busca en entorno
      (lit-exp (datum) datum)           ; Literal: retorna el número
      (true-exp () #t)                  ; True: retorna #t
      (false-exp () #f)                 ; False: retorna #f
      (prim-exp (prim lrands)           ; Primitiva: evalúa argumentos y aplica
        (let ((values (map (lambda (x) (eval-expression x env)) lrands)))
          (apply-primitive prim values)))
      (if-exp (cond-exp true-exp false-exp)
        (let ((cond-value (eval-expression cond-exp env)))
          (if (boolean? cond-value)
              (if cond-value
                  (eval-expression true-exp env)
                  (eval-expression false-exp env))
              (eopl:error "The condition must be a boolean " cond-exp))))
      (let-exp (lid lexpr expr)
        (let ((vexpr (map (lambda (x) (eval-expression x env)) lexpr)))
          (eval-expression expr (extend-env lid (list->vector vexpr) env))))
      (proc-exp (types lid exp)
        (closure lid exp env))
      (app-exp (rator rands)
        (let ((procv (eval-expression rator env))
              (vrands (map (lambda (x) (eval-expression x env)) rands)))
          (if (and (procVal? procv) (= (length (procVal->lid procv)) (length vrands)))
              (cases procVal procv
                (closure (lid exp old-env)
                  (eval-expression exp (extend-env lid (list->vector vrands) old-env))))
              (eopl:error "Not a procedure or incorrect of number of args"))))
      ;; Manejo de expresiones letrec en eval-expression
      (letrec-exp (type procnames lltypes llargs bodies expn)
        ; Componentes de la expresión letrec
        (eval-expression expn ; Expresión del cuerpo principal (después de "in")
          (extend-recursively-env ; Extender el ambiente con definiciones recursivas
            procnames            ; Lista de nombres de procedimientos
            llargs               ; Lista de listas de argumentos (parámetros formales)
            bodies               ; Lista de cuerpos de los procedimientos
            env)))               ; Ambiente actual (base para la extensión)
      (begin-exp (exp lexp)
        (letrec ((evaluation
                   (lambda (lexp)
                     (cond
                       [(null? (cdr lexp)) (eval-expression (car lexp) env)]
                       [else (begin (eval-expression (car lexp) env) (evaluation (cdr lexp)))]))))
          (evaluation (cons exp lexp))))
      (set-exp (id exp)
        (begin
          (setref! (apply-env-ref env id) (eval-expression exp env))
          (void)))
      (else "Not implemented yet")))) ; Default (no debería ocurrir)

###
La función `operation` es una utilidad genérica que aplica una función binaria acumulativa a una lista de valores. `apply-primitive` toma una primitiva (como `+` o `and`) y una lista de valores, luego ejecuta la operación correspondiente. Maneja operaciones aritméticas (suma, resta, multiplicación, división), lógicas (AND, OR) y comparaciones (`<`, `<=`, `>`, `>=`, `==`, `!=`).

; Operación genérica: aplica una función binaria acumulativa a una lista
; operation: (T,T)->T, lista de valores, T -> value
(define operation
  (lambda (f lst acc)
    (cond
      [(null? lst) acc] ; Caso base: retorna acumulador
      [else (operation f (cdr lst) (f (car lst) acc))]))) ; Aplica f y acumula

; Aplicar primitiva: ejecuta la operación correspondiente sobre los valores
; apply-primitive: primitive, lista valores -> value
(define apply-primitive
  (lambda (prim values)
    (cases primitive prim
      (add-prim () (operation + values 0)) ; Suma acumulativa
      (sub-prim () (- (car values) (operation + (cdr values) 0))) ; Resta (first - sum(rest))
      (prod-prim () (operation * values 1)) ; Producto acumulativo
      (div-prim () (/ (car values) (operation * (cdr values) 1))) ; División (first / product(rest))
      (and-prim () (operation (lambda (a b) (and a b)) values #t)) ; AND acumulativo
      (or-prim () (operation (lambda (a b) (or a b)) values #f)) ; OR acumulativo
      (less-prim () (< (car values) (cadr values)))     ; < (2 args)
      (lesseq-prim () (<= (car values) (cadr values)))   ; <= (2 args)
      (more-prim () (> (car values) (cadr values)))     ; > (2 args)
      (moreeq-prim () (>= (car values) (cadr values)))   ; >= (2 args)
      (eq-prim () (= (car values) (cadr values)))       ; == numérico (2 args)
      (neq-prim () (not (= (car values) (cadr values))))))) ; != numérico (2 args)

Las funciones `expand-type-expression` y `expand-type-expressions` se encargan de transformar las expresiones de tipos que provienen de la gramática (`type-exp`) en la representación interna de tipos (`type`) utilizada por el chequeador de tipos.

;expand-type-expression: expresion de tipos -> tipo
(define expand-type-expression
  (lambda (texp)
    (cases type-exp texp
      (int-exp () int-type)
      (bool-exp () bool-type)
      (proc-type-exp (targs tresult)
        (proc-type (expand-type-expressions targs) (expand-type-expression tresult))))))

(define expand-type-expressions
  (lambda (texps)

###
`init-tenv` es el entorno de tipos inicial, que mapea identificadores a sus tipos predefinidos. `apply-tenv` (una versión de `apply-env` adaptada para entornos de tipos) se utiliza para buscar el tipo de una variable dentro de este entorno.

;Type enviroment
(define init-tenv
  (extend-env
    '(x y z f)
    (list->vector (list int-type int-type int-type (proc-type (list int-type int-type int-type) int-type)))
    (extend-env
      '(a b c)
      (list->vector (list int-type int-type int-type))
      (empty-env))))

; apply-env (typo en el nombre, debería ser apply-tenv): environment -> type
; Este `apply-env` es una versión para el Type Environment (`tenv`), no el `env` de valores.
(define apply-tenv
  (lambda (env var)
    (cases environment env
      (empty-env ()
        (eopl:error "Variable not found" var)) ; Error si no se encuentra
      (extend-env (lid vec old-env)
        (letrec ((search-var ; Búsqueda recursiva en la lista de bindings
                    (lambda (lid lval [pos 0])
                      (cond
                        [(null? lid) (apply-tenv old-env var)] ; Buscar en entorno anterior
                        [(equal? (car lid) var) (a-ref pos vec)]   ; Voy a generar una referencia cuando encuentro la variable
                        [else (search-var (cdr lid) vec (+ pos 1))]))) ; Continuar búsqueda
          (search-var lid vec)))
      (extend-recursively-env (procnames llargs bodies old-env)
        (letrec ((search-proc
                    (lambda (procs args bodies)
                      (cond
                        [(null? procs) (apply-env-ref old-env var)]
                        [(eqv? (car procs) var) (a-ref 0 (list->vector (list (closure (car args) (car bodies) env))))]
                        [else (search-proc (cdr procs) (cdr args) (cdr bodies))]))))
          (search-proc procnames llargs bodies)))))))

###
La función `check-equal-type!` verifica si dos tipos son iguales, lanzando un error si no coinciden. `type-to-external-form` y `arg-types-to-external-form` convierten las representaciones internas de los tipos a un formato legible para el usuario, útil en los mensajes de error.

;check-equal-type: type, type -> type
(define check-equal-type!
  (lambda (t1 t2 exp)
    (if (not (equal? t1 t2))
        (eopl:error 'type-check "Types didn't match ~s != ~s in ~%~s"
          (type-to-external-form t1)
          (type-to-external-form t2)
          exp)
        t1)))

(define type-to-external-form
  (lambda (t)
    (cases type t
      (atomic-type (name) name)
      (proc-type (targs tresult)
        (append (arg-types-to-external-form targs)
          '(-->)
          (list (type-to-external-form tresult)))))))

(define arg-types-to-external-form
  (lambda (ty)
    (cond
      [(null? ty) '()]
      [(null? (cdr ty)) (list (type-to-external-form (car ty)))]
      [else (append (list (type-to-external-form (car ty)))
              (list "*")
              (arg-types-to-external-form (cdr ty)))])))

###
La función `type-of-expression` es el corazón del chequeador de tipos estático. Determina el tipo de una expresión dada un entorno de tipos. Cubre:
*   Literales (`lit-exp`, `true-exp`, `false-exp`): sus tipos son directos (`int-type`, `bool-type`).
*   Variables (`var-exp`): el tipo se busca en el entorno de tipos.
*   Condicionales (`if-exp`): asegura que la condición sea booleana y que ambas ramas (`true-exp`, `false-exp`) tengan el mismo tipo.
*   Procedimientos (`proc-exp`): el tipo se infiere usando `type-of-proc-exp`.
*   Aplicaciones (`app-exp`): el tipo se infiere usando `type-of-application`.
*   `let-exp`: el tipo del cuerpo se verifica en un entorno extendido con los tipos de las variables locales.
*   Primitivas (`prim-exp`): el tipo se infiere usando `type-of-primitive` y `type-of-application`.
*   `letrec-exp`: el tipo se infiere usando `type-of-letrec` para manejar procedimientos recursivos.
*   Asignaciones (`set-exp`): se asume un tipo de retorno (`int-type`).
*   Bloques `begin` (`begin-exp`): el tipo se infiere usando `begin-type-exp`.

;type-of-expression: expression -> type
(define type-of-expression
  (lambda (exp tenv)
    (cases expression exp
      (lit-exp (d) int-type)
      (true-exp () bool-type)
      (false-exp () bool-type)
      (var-exp (id) (deref (apply-tenv tenv id)))
      (if-exp (test-exp true-exp false-exp)
        (let ((test-type (type-of-expression test-exp tenv))
              (true-type (type-of-expression true-exp tenv))
              (false-type (type-of-expression false-exp tenv)))
          (begin
            (check-equal-type! test-type bool-type test-exp)
            (check-equal-type! true-type false-type exp)
            true-type))) ; false-type also works here as they are checked to be equal
      (proc-exp (targs args body)
        (type-of-proc-exp targs args body tenv))
      (app-exp (rator rands)
        (let ((trator (type-of-expression rator tenv))
              (trands (map (lambda (x) (type-of-expression x tenv)) rands)))
          (type-of-application trator trands rator rands exp)))
      (let-exp (lid lval exp)
        (type-of-expression exp
          (extend-env lid
            (list->vector (map (lambda (x) (type-of-expression x tenv)) lval))
            tenv)))
      (prim-exp (prim rands)
        (type-of-application
          (type-of-primitive prim rands)
          (map (lambda (x) (type-of-expression x tenv)) rands)
          prim rands exp))
      (letrec-exp (typesproc procnames ttargs args bodies exp)
        (type-of-letrec exp typesproc procnames ttargs args bodies tenv))
      (set-exp (id exp) int-type) ; Assuming 'set' expressions always return int-type, might be simplified/incorrect
      (begin-exp (exp exps)
        (begin-type-exp (cons exp exps) tenv)))))

###
`type-of-proc-exp` determina el tipo de una definición de procedimiento (`proc`). Primero expande los tipos de los argumentos y luego infiere el tipo de retorno evaluando el cuerpo del procedimiento en un entorno extendido con los tipos de sus argumentos.

`type-of-application` verifica si una aplicación de procedimiento es correcta. Compara la longitud y los tipos de los argumentos reales (`trands`) con los tipos esperados por el procedimiento (`args-types`) obtenidos de su tipo (`trator`).

;type-of-proc-exp list of types, list of idenfier, exp => type
(define type-of-proc-exp
  (lambda (targs args body tenv)
    (let* ((args-types (expand-type-expressions targs))
           (result-type (type-of-expression body (extend-env args (list->vector args-types) tenv))))
      (proc-type args-types result-type))))

(define type-of-application
  (lambda (trator trands rator rands exp)
    (cases type trator
      (proc-type (args-types result-type)
        (if (= (length args-types) (length trands))
            (begin
              (for-each check-equal-type! trands args-types rands)
              result-type)
            (eopl:error 'type-of-application "Length of the args mismatch ~s ~s" args-types trands)))
      (else
        (eopl:error 'type-of-expression "Rator isn't a proc-type ~s is a type ~s" rator (type-to-external-form trator))))))

###
`type-of-primitive` define los tipos de los procedimientos para las operaciones primitivas del lenguaje. Para cada primitiva (suma, resta, AND, comparaciones, etc.), especifica el `proc-type` que describe los tipos de sus argumentos y el tipo de su valor de retorno.

;; TIpos de los procedimientos
(define type-of-primitive
  (lambda (prim rands)
    (cases primitive prim
      (add-prim () (proc-type (map (lambda (x) int-type) rands) int-type))
      (sub-prim () (proc-type (map (lambda (x) int-type) rands) int-type))
      (prod-prim () (proc-type (map (lambda (x) int-type) rands) int-type))
      (div-prim () (proc-type (map (lambda (x) int-type) rands) int-type))
      (and-prim () (proc-type (map (lambda (x) bool-type) rands) bool-type))
      (or-prim () (proc-type (map (lambda (x) bool-type) rands) bool-type))
      (less-prim () (proc-type (list int-type int-type) bool-type))
      (lesseq-prim () (proc-type (list int-type int-type) bool-type))
      (more-prim () (proc-type (list int-type int-type) bool-type))
      (moreeq-prim () (proc-type (list int-type int-type) bool-type))
      (eq-prim () (proc-type (list int-type int-type) bool-type))
      (neq-prim () (proc-type (list int-type int-type) bool-type)))))

###
`type-of-letrec` maneja el chequeo de tipos para definiciones de procedimientos recursivos. Construye un entorno de tipos que incluye los tipos de los procedimientos recursivos antes de verificar los cuerpos de estos procedimientos y la expresión principal. `begin-type-exp` chequea los tipos de una secuencia de expresiones, retornando el tipo de la última expresión.

;;Procedimientos recursivos
(define type-of-letrec
  (lambda (exp tprocs procnames ttargs args bodies tenv)
    (let* ((arg-types (map (lambda (x) (expand-type-expressions x)) ttargs))
           (result-types (map (lambda (x) (expand-type-expression x)) tprocs))
           (the-proc-types (map proc-type arg-types result-types))
           (the-env-body (extend-env procnames (list->vector the-proc-types) tenv)))
      (for-each
        (lambda (ids args-types body result-type)
          (check-equal-type!
            (type-of-expression body (extend-env ids (list->vector args-types) the-env-body))
            result-type
            body))
        args arg-types bodies result-types)
      (type-of-expression exp the-env-body))))

;;begin-type-exp: lista exp -> type
(define begin-type-exp
  (lambda (lexp tenv)
    (cond
      [(null? (cdr lexp)) (type-of-expression (car lexp) tenv)]
      [else
        (begin
          (type-of-expression (car lexp) tenv)
          (begin-type-exp (cdr lexp) tenv))])))

###
Se definen dos entornos iniciales de ejecución: `env0` establece algunas variables básicas (`a`, `b`, `c`) con valores numéricos. `init-env` extiende `env0` con variables adicionales (`x`, `y`, `z`) y un procedimiento predefinido `f` que realiza una suma. Estos entornos proporcionan un estado inicial para la evaluación de programas.

; Entorno inicial predefinido (variables x,y,z y a,b,c con valores numéricos)
(define env0
  (extend-env
    '(a b c)
    (list->vector '(4 5 6))
    (empty-env)))

(define init-env
  (extend-env
    '(x y z f)
    (list->vector
      (list 1 2 3
        (closure
          '(x y)
          (prim-exp (add-prim) (list (var-exp 'x) (var-exp 'y)))
          env0)))
    env0))

###
Esta sección configura y lanza el intérprete interactivo. `sllgen:make-rep-loop` crea un bucle de lectura-evaluación-impresión (REPL) con un prompt "-->". Utiliza `eval-program` para procesar el input y `sllgen:make-stream-parser` para analizar el programa. Finalmente, la llamada `(interpreter)` inicia el REPL.

; Intérprete interactivo: loop de lectura-evaluación-impresión
(define interpreter
  (sllgen:make-rep-loop
    "-->"             ; Prompt
    eval-program      ; Función de evaluación
    (sllgen:make-stream-parser lexical grammar))) ; Parser para input stream

; Iniciar el intérprete
(interpreter)
